// To print the elements on the basis of their frequencies arranged in decrement order (if frequency is same then the smallest element is printed respectively)
#include<bits/stdc++.h>
using namespace std;
int main(){
    int n;
    cin>>n;
    // Main Array
    int arr[n];
    for(int i=0;i<n;i++){
        cin>>arr[i];
    }
    int arr1[n];
    int arr2[n];
    int ele[n];
    int size=0;
    int found;
    int count;
    // creating 3 new arrays
    // arr1 -> to store the frequency and then sort
    // arr2 -> to store the frequency but no sort ; to locate the position of ele
    // ele -> to match the element with their frequency
    for(int j=0;j<n;j++){
        found=0;
        for(int k=0;k<size;k++){
            if(ele[k]==arr[j]){
                found=1;
                break;
                
            }
        }
        if(found==0){
            count=1;
            for(int i=j+1;i<n;i++){
                if(arr[j]==arr[i]){
                    count++;
                }
            }
            arr2[size]=count;
            arr1[size]=count;
            ele[size]=arr[j];
            size++;
        }
    }
// To ensure this condition If two elements have the same frequency, they should be sorted in ascending order of their values.
    for(int k=0;k<size;k++){
        for(int j=0;j<size-1;j++){
            if(ele[j]>=ele[j+1]){
                int temp=ele[j];
                ele[j]=ele[j+1];
                ele[j+1]=temp;
                int temp2=arr2[j];
                arr2[j]=arr2[j+1];
                arr2[j+1]=temp2;
            }
        }
    }
    // Sorting the array of frequency of elements
    for(int k=0;k<size;k++){
        for(int j=0;j<size-1;j++){
            if(arr1[j]<arr1[j+1]){
                int temp=arr1[j];
                arr1[j]=arr1[j+1];
                arr1[j+1]=temp;
            }
        }
    }
    //Creating new array to check if that indice is covered or not
    int ind[size];
    int size2=0;
    int f;
    for(int i=0;i<size;i++){
        for(int j=0;j<size;j++){
            if(arr1[i]==arr2[j]){
                f=0;
                for(int k=0;k<size2;k++){
                    if(ind[k]==j){
                        f=1;
                        break;
                    }
                }
                if(f==0){
                ind[size2++]=j;
                for(int k=0;k<arr1[i];k++){
                    cout<<ele[j]<<" ";
                }
                break;
                }
            }
        }
    }
    return 0;
}
