#include<iostream>
using namespace std;

class Node{
    public:
    int data;
    Node* next;
    Node(int val){
        this->data=val;
        this->next=NULL;
    }
};

void insertathead(Node* &head,int val){
    Node* temp=new Node(val);
    temp->next=head;
    head=temp;
}

void insertattail(Node* &tail,int val){
    Node* temp=new Node(val);
    tail->next=temp;
    tail=temp;
}

void print(Node* &head){
    Node* temp=head;
    while(temp!=NULL){
        cout<<temp->data<<" ";
        temp=temp->next;
    }
    cout<<endl;
}

void insertatgivenposition(Node* &head,Node* &tail,int position,int val){
    if(position==1){
        /*
        Node* temp=new Node(val);
        temp->next=head;
        head=temp;
        */
        insertathead(head,val);
        return;
    }
    
    // We have taken care of insertathead but what if the user gives position at end then element would be added but the tail will not be updated
    
    Node* temp=head;
    int count=1;
    while(count<position-1){
        temp=temp->next;
        count++;
    }
    if(temp->next==NULL){
        insertattail(tail,val);
        return;
    }
    Node* toinsert=new Node(val);
    toinsert->next=temp->next;
    temp->next=toinsert;
}

int main(){
    Node* node2=new Node(20);
    Node* head=node2;
    Node* tail=node2;
    insertathead(head,10);
    insertattail(tail,30);
    insertatgivenposition(head,3,25);
    print(head);


    return 0;
}
