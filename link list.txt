#include <iostream>
using namespace std;

class Node{
  public:
  int data;
  Node* next;
  Node(int val){
      this->data=val;
      this->next=NULL;
  }
};

void insertathead(Node* &head,int val){
    Node* temp=new Node(val);
    temp->next=head;
    head=temp;
}

void insertattail(Node* &tail,int val){
    Node* temp=new Node(val);
    tail->next=temp;
    tail=temp;
}

void print(Node* &head){
    Node* &temp=head;
    while(temp!=NULL){
        cout<<temp->data<<" ";
        temp=temp->next;
    }
    cout<<endl;
}

int main()
{
    Node* node1=new Node(20);
    Node* head=node1;
    Node* tail=node1;
    insertathead(head,10);
    insertattail(tail,30);
    print(head);
    return 0;
}
